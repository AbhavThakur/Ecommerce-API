Users

1. register

Endpoint:

Method: POST
Type: RAW
URL: {{url}}/{{version}}/users/register

Body:

{
  "fullName":"Rohan",
  "email":"Rohan@gmail.com",
  "password":"12345"
}

2. login

Endpoint:

Method: POST
Type: RAW
URL: {{url}}/{{version}}/users/login

Body:

{
  "email":"Rohan@gmail.com",
  "password":"12345"
}

3. GET Profile

Endpoint:

Method: GET
Type:
URL: {{url}}/{{version}}/users/profile

4. Update Shipping Address

Endpoint:

Method: PUT
Type: RAW
URL: {{url}}/{{version}}/users/update/shipping

Body:

{
  "fullName":"hello",
  "email":"Nate",
  "address":"Himachal",
  "city":"Mandi",
  "postalCode":"173026",
  "country":"India",
  "province":"Near Village"
}

5. Success

Endpoint:

Method: GET
Type:
URL: {{url}}/{{version}}/success
